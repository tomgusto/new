Animated Splash Screen Examples

1. Pure CSS Animation

Lightweight and simple animations using only CSS.

HTML 
<div class="splash-screen">
  <div class="logo"></div>
</div>
CSS 
.splash-screen {
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  display: flex;
  justify-content: center;
  align-items: center;
  background: #ffffff;
  z-index: 1000;
  animation: fadeOut 0.5s ease-out 2.5s forwards;
}

.logo {
  width: 100px;
  height: 100px;
  background: #3498db;
  border-radius: 50%;
  animation: 
    pulse 2s infinite,
    rotate 4s linear infinite;
}

@keyframes pulse {
  0% { transform: scale(0.95); }
  50% { transform: scale(1.05); }
  100% { transform: scale(0.95); }
}

@keyframes rotate {
  from { transform: rotate(0deg); }
  to { transform: rotate(360deg); }
}

@keyframes fadeOut {
  to { opacity: 0; visibility: hidden; }
}

2. Lottie Animation

High-quality animations exported from After Effects. Lightweight and scalable.

HTML 
<div class="splash-screen">
  <div id="lottie-animation"></div>
</div>

<script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.9.6/lottie.min.js"></script>
JavaScript 
const animation = lottie.loadAnimation({
  container: document.getElementById('lottie-animation'),
  renderer: 'svg',
  loop: true,
  autoplay: true,
  path: 'animation.json' // Path to your Lottie JSON
});

// Hide splash screen when animation completes
animation.addEventListener('DOMLoaded', function() {
  setTimeout(() => {
    document.querySelector('.splash-screen').style.opacity = 0;
    setTimeout(() => {
      document.querySelector('.splash-screen').style.display = 'none';
    }, 500);
  }, 2500);
});

3. GSAP Animation

Professional-grade animations with precise control and excellent performance.

HTML 
<div class="splash-screen">
  <div class="logo"></div>
</div>

<script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.11.4/gsap.min.js"></script>
JavaScript 
// Create timeline
const tl = gsap.timeline({ defaults: { ease: "power2.inOut" } });

tl.from(".logo", {
  duration: 0.8,
  scale: 0.5,
  opacity: 0,
  y: 50,
  rotation: -180
})
.to(".logo", {
  duration: 0.8,
  scale: 1.2,
  y: -20,
  rotation: 0,
  ease: "back.out(1.7)"
})
.to(".logo", {
  duration: 0.5,
  scale: 1,
  y: 0
});

// Hide splash screen
tl.to(".splash-screen", {
  duration: 0.5,
  opacity: 0,
  onComplete: () => {
    document.querySelector(".splash-screen").style.display = "none";
  }
}, "+=1");

4. SVG Animation

Resolution-independent vector animations that scale perfectly on any device.

HTML 
<div class="splash-screen">
  <svg class="logo" viewBox="0 0 100 100">
    <circle class="circle" cx="50" cy="50" r="40" />
  </svg>
</div>
CSS 
.splash-screen {
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  display: flex;
  justify-content: center;
  align-items: center;
  background: #ffffff;
  z-index: 1000;
}

.logo {
  width: 100px;
  height: 100px;
}

.circle {
  fill: #2ecc71;
  transform-origin: center;
  animation: pulse 2s ease-in-out infinite;
}

@keyframes pulse {
  0% { transform: scale(1); opacity: 0.7; }
  50% { transform: scale(1.1); opacity: 1; }
  100% { transform: scale(1); opacity: 0.7; }
}

5. Three.js 3D Animation

For advanced 3D animations (more resource-intensive).

HTML 
<div class="splash-screen">
  <div id="threejs-container"></div>
</div>

<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
JavaScript 
// Initialize Three.js
const container = document.getElementById('threejs-container');
const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);
const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

renderer.setSize(container.clientWidth, container.clientHeight);
container.appendChild(renderer.domElement);

// Create a cube
const geometry = new THREE.BoxGeometry();
const material = new THREE.MeshBasicMaterial({ 
  color: 0x3498db,
  wireframe: true
});
const cube = new THREE.Mesh(geometry, material);
scene.add(cube);

camera.position.z = 2;

// Animation loop
function animate() {
  requestAnimationFrame(animate);
  
  cube.rotation.x += 0.01;
  cube.rotation.y += 0.01;
  
  renderer.render(scene, camera);
}

// Handle window resize
window.addEventListener('resize', () => {
  camera.aspect = container.clientWidth / container.clientHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(container.clientWidth, container.clientHeight);
});

// Hide splash screen after delay
setTimeout(() => {
  document.querySelector('.splash-screen').style.opacity = 0;
  setTimeout(() => {
    document.querySelector('.splash-screen').style.display = 'none';
  }, 500);
}, 3000);

// Start animation
animate();
